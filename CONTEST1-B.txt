#include <stdio.h>
#include <math.h>

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        long long k;
        scanf("%lld", &k);
        long long n = (long long)((sqrt(8.0 * k + 1) - 1) / 2);
        if (n * (n + 1) / 2 < k)
            n++;
        printf("%lld\n", n);
    }
    return 0;
}
